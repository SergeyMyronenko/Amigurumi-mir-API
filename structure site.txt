ecommerce-backend/
│
├── config/
│   ├── db.js              // Підключення до бази даних MongoDB
│   └── config.js          // Конфігураційні змінні середовища
│
├── controllers/
│   ├── userController.js  // Контролери для обробки запитів користувачів
│   ├── productController.js  // Контролери для обробки запитів товарів
│   └── orderController.js // Контролери для обробки запитів замовлень
│
├── middlewares/
│   ├── authMiddleware.js  // Перевірка авторизації користувача
│   └── errorHandler.js    // Обробка помилок
│
├── models/
│   ├── User.js            // Схема для колекції користувачів
│   ├── Product.js         // Схема для колекції товарів
│   └── Order.js           // Схема для колекції замовлень
│
├── routes/
│   ├── userRoutes.js      // Роутинг для користувачів (реєстрація, вхід тощо)
│   ├── productRoutes.js   // Роутинг для товарів (додавання, перегляд тощо)
│   └── orderRoutes.js     // Роутинг для замовлень (оформлення, перегляд)
│
├── utils/
│   └── generateToken.js   // Функція для генерації токенів авторизації
│
├── .env                   // Файл змінних середовища (наприклад, MongoDB URI, секрети JWT)
├── .gitignore             // Файли, які ігноруватимуться у Git
├── package.json           // Інформація про залежності та скрипти
└── server.js              // Основний файл серверу, що ініціалізує додаток
